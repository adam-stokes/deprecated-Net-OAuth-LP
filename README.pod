NAME
    Net::OAuth::LP::Client - Launchpad.net Client routines

SYNOPSIS
        use Net::OAuth::LP::Client;
        my $lp = Net::OAuth::LP::Client->new;
        $lp->login_with_creds;

METHODS
  login_with_creds
        use Net::OAuth::LP::Client;

        my $lp = Net::OAuth::LP::Client->new;
        $lp->consumer_key('my-lp-app');

        # Authorize yourself
        $lp->login_with_creds;
